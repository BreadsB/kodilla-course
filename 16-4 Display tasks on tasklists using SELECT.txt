SELECT I2.NAME, I.SUMMARY AS TASK_NAME
FROM issues AS I
JOIN issueslist I2 on I.ISSUESLIST_ID = I2.ID;


SELECT U.FIRSTNAME, U.LASTNAME, I.SUMMARY AS TASK_NAME
FROM USERS AS U
JOIN issues I on U.ID = I.USER_ID_ASSIGNEDTO
ORDER BY U.LASTNAME;


SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASKS_QUANTITY
FROM USERS AS U
JOIN issues i on U.ID = i.USER_ID_ASSIGNEDTO
GROUP BY U.ID
ORDER BY U.LASTNAME;
